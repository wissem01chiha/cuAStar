181__device__ float fatomicMin(float *addr, float value) {
182 float old = *addr, assumed;
183 if (old <= value) return old;
184 do {
185 assumed = old;
186 old = atomicCAS((unsigned int*)addr, __float_as_int(assumed),
187 __float_as_int(value < assumed ? value : assumed));
188 } while (old != assumed);
189 return old;
190}
191
192__device__ float fatomicMax(float *addr, float value) {
193 float old = *addr, assumed;
194 if (old >= value) return old;
195 do {
196 assumed = old;
197 old = atomicCAS((unsigned int*)addr, __float_as_int(assumed),
198 __float_as_int(value > assumed ? value : assumed));
199 } while (old != assumed);
200 return old;
201}

481void drawFilledCircle(unsigned char* image, int width, int height, const Circle* circle,
482 const Color* color) {
483 int cx = circle->x;
484 int cy = circle->y;
485 int radius = circle->radius;
486
487 for (int y = (cy - radius > 0 ? cy - radius : 0); y <= (cy + radius < height ? cy +
488 radius : height - 1); ++y) {
489 for (int x = (cx - radius > 0 ? cx - radius : 0); x <= (cx + radius < width ? cx
490 + radius : width - 1); ++x) {
491 int dx = x - cx;
492 int dy = y - cy;
493 if (dx * dx + dy * dy <= radius * radius) {
494 int index = (y * width + x) * 3;
495 image[index] = color->r;
496 image[index + 1] = color->g;
497 image[index + 2] = color->b;
498 }
499 }
500 }

549template <typename NodeType>
550__global__ void computePointCloudBoundBox(const NodeType* h_arrayNodes, int N,
551 float* minX, float* maxX, float* minY, float* maxY){
552
553 __shared__ float localMinX, localMaxX, localMinY, localMaxY;
554
555 int idx = blockIdx.x * blockDim.x + threadIdx.x;
556
557 if( threadIdx.x == 0){ // for each block intilize eeach thred has it owen copy
558 localMinX = FLT_MAX;
559 localMaxX = -FLT_MAX;
560 localMinY = FLT_MAX;
561 localMaxY = -FLT_MAX;
562 }
563 __syncthreads();
564
565 if (idx < N){
566
567 NodeType node = h_arrayNodes[idx];
568
569 fatomicMax(&localMaxX, node.x);
570 fatomicMin(&localMinX, node.x);
571 fatomicMax(&localMaxY, node.y);
572 fatomicMin(&localMinY, node.y);
573 }
574 __syncthreads();
575
576 if (threadIdx.x == 0) {
577 fatomicMin(minX, localMinX);
578 fatomicMax(maxX, localMaxX);
579 fatomicMin(minY, localMinY);
580 fatomicMax(maxY, localMaxY);
581 }
582}

590template <typename NodeType>
591HOST_FUN void array2PointCloudImg(const NodeType* h_arrayNodes, int numNodes,
592 const char* pngFilePath,double scaleFactor=100,
593 double radiusRatio = 0.01) {
594
595 float h_minX = FLT_MAX, h_maxX = -FLT_MAX, h_minY = FLT_MAX, h_maxY = -FLT_MAX;
596
597 float *d_minX, *d_maxX, *d_minY, *d_maxY;
598 cudaMalloc(&d_minX, sizeof(float));
599 cudaMalloc(&d_maxX, sizeof(float));
600 cudaMalloc(&d_minY, sizeof(float));
601 cudaMalloc(&d_maxY, sizeof(float));
602
603 cudaMemcpy(d_minX, &h_minX, sizeof(float), cudaMemcpyHostToDevice);
604 cudaMemcpy(d_maxX, &h_maxX, sizeof(float), cudaMemcpyHostToDevice);
605 cudaMemcpy(d_minY, &h_minY, sizeof(float), cudaMemcpyHostToDevice);
606 cudaMemcpy(d_maxY, &h_maxY, sizeof(float), cudaMemcpyHostToDevice);
607
608 NodeType* d_arrayNodes;
609 cudaMalloc(&d_arrayNodes, numNodes * sizeof(NodeType));
610 cudaMemcpy(d_arrayNodes, h_arrayNodes, numNodes * sizeof(NodeType), cudaMemcpyHostToDevice);
611
612 int blocksNum = (numNodes + threadsPerBlock - 1) / threadsPerBlock;
613 computePointCloudBoundBox<NodeType><<<blocksNum, threadsPerBlock>>>(
614 d_arrayNodes, numNodes, d_minX, d_maxX, d_minY, d_maxY);
615
616 cudaMemcpy(&h_minX, d_minX, sizeof(float), cudaMemcpyDeviceToHost);
617 cudaMemcpy(&h_maxX, d_maxX, sizeof(float), cudaMemcpyDeviceToHost);
618 cudaMemcpy(&h_minY, d_minY, sizeof(float), cudaMemcpyDeviceToHost);
619 cudaMemcpy(&h_maxY, d_maxY, sizeof(float), cudaMemcpyDeviceToHost);
620
621 cudaFree(d_minX);
622 cudaFree(d_maxX);
623 cudaFree(d_minY);
624 cudaFree(d_maxY);
625 cudaFree(d_arrayNodes);
626
627 int* centersX = new int[numNodes];
628 int* centersY = new int[numNodes];
629 unsigned char* colorsR = new unsigned char[numNodes];
630 unsigned char* colorsG = new unsigned char[numNodes];
631 unsigned char* colorsB = new unsigned char[numNodes];
632
633 double r = (h_maxY - h_minY)/(h_maxX - h_minX);
634 int width = static_cast<int>(scaleFactor * (h_maxX - h_minX));
635 int height = static_cast<int>( r * width);
636
637 for (int i = 0; i < numNodes; ++i) {
638 centersX[i] = static_cast<int>((h_arrayNodes[i].x - h_minX)/(h_maxX - h_minX) * width );
639 centersY[i] = static_cast<int>((h_arrayNodes[i].y - h_minY)/(h_maxY - h_minY) * height );

641 colorsG[i] = static_cast<unsigned char>(h_arrayNodes[i].g);
642 colorsB[i] = static_cast<unsigned char>(h_arrayNodes[i].b);
643 }
644
645 savePointCloudImage(pngFilePath, width+1, height+1, centersX,
646 centersY, colorsR, colorsG, colorsB, static_cast<int>(numNodes),
647 radiusRatio);
648
649 delete[] centersX;
650 delete[] centersY;
651 delete[] colorsR;
652 delete[] colorsG;
653 delete[] colorsB;
654}

748template <typename T>
749__global__ void enumerationSort(T * a, int N, T * b){
750
751 int cnt = 0;
752 int tid = threadIdx.x;
753 int ttid = blockIdx.x * threadsPerBlock + tid;
754 T val = a[ttid];
755 __shared__ T cache[threadsPerBlock];

813DEVICE_FUN void updateNearest(NodeType* nearestNodes, T* distances,
814 const NodeType& node, const NodeType& otherNode, int k) {
815
816 for (int i = 0; i < k; ++i) {
817 if (nearestNodes[i].isEqual(otherNode,static_cast<T>(1e-5))) {
818 return;
819 }
820 }
821 T dist = node.distanceTo(otherNode);
822 for (int i = 0; i < k; ++i) {
823 if (dist < distances[i]) {
824 for (int j = k - 1; j > i; --j) {
825 distances[j] = distances[j - 1];
826 nearestNodes[j] = nearestNodes[j - 1];
827 }
828 distances[i] = dist;
829 nearestNodes[i] = otherNode;
830 break;
831 }
832 }
